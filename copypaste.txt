javascript:(() => {
  const now = new Date();
  console.log('Current Date and Time (Local):', now.toString());
  console.log('Current Date and Time (UTC):', now.toUTCString());
  const rawEpoch = Math.floor(now.getTime() / 1000);
  console.log('Raw Epoch (UTC Seconds):', rawEpoch);
  const adjustedEpoch = rawEpoch - 18000;
  console.log('Adjusted Epoch (EST):', adjustedEpoch);
  const timeStep = Math.floor(adjustedEpoch / 30);
  console.log('Time Step (TOTP Interval, EST):', timeStep);
  alert(`Local Time: ${now.toString()}\nUTC Time: ${now.toUTCString()}\nRaw Epoch (UTC): ${rawEpoch}\nAdjusted Epoch (EST): ${adjustedEpoch}\nTOTP Time Step (EST): ${timeStep}`);
})();
