(async () => {
  try {
    // URL of your Flask server
    const url = "http://127.0.0.1:5000/generate-otp"; // Change to your Flask server URL

    // Fetch OTP from Flask server
    const response = await fetch(url);
    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

    const data = await response.json();
    console.log("Generated OTP:", data.otp);

    // Display the OTP
    alert(`Generated OTP: ${data.otp}`);
  } catch (error) {
    console.error("Error fetching OTP:", error);
    alert("Failed to fetch OTP. Check the console for details.");
  }
})();
