=LET(
  text, C2,
  apt, E2,
  suffixes, {" ST ", " RD ", " AVE ", " BLVD ", " LN ", " DR ", " CT ", " PKWY ", " WAY ", " PL "},

  lastSuffixPos,
    MAX(IFERROR(FIND(suffix, UPPER(text) & " "), 0) + LEN(suffix) - 1),

  cityStartPos,
    IF(apt <> "",
      FIND(apt, text) + LEN(apt) + 1,
      lastSuffixPos + 2
    ),

  city,
    TRIM(MID(text, cityStartPos, LEN(text))),

  city
)
